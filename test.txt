function print(){

	if (!$('input:checkbox[name="list_select"]').is(":checked")) {
		alert("선택된 항목이 없습니다.");
		return;
	}

	var selectSection4Print = "";
	
	$('input:checkbox[name="list_select"]').each(function(index, item){
		if($(this).is(":checked")){
			selectSection4Print += "#inquiry"+$(this).val()+",";
		}
	});
	selectSection4Print = selectSection4Print.substring(0,selectSection4Print.length-1);
	
	$(selectSection4Print).show().printThis();
	setTimeout(function(){
		$(selectSection4Print).hide();
		$($(".check_wrap").find(".link.on").attr("href")).show();
	},1000);
}

function checkAll(type){
	if($("#checkAll_"+type).is(":checked") == true){
		$(".link:visible").each(function(index, item){
			if($(this).closest("div").find("input").attr("duty") == type){
				$(this).closest("div").find("input").prop("checked", true);
			}
		});
	}else{
		$(".link:visible").each(function(index, item){
			if($(this).closest("div").find("input").attr("duty") == type){
				$(this).closest("div").find("input").prop("checked", false);
			}
		});
	}
}

<p class="guide_txt">※ 생물화학적 산소요구량(BOD, Biochemical Oxygen Demand)란?<br/>호기성 미생물이 일정 기간 동안 물 속에 있는 유기물을 분해할 때 사용하는 산소의 양을 의미하며, COD와 함께 물의 오염된 정도를 표시하는 지표로 사용됨</p>
<p class="guide_txt">※ 화학적 산소요구량(COD, Chemical Oxygen Demand)란?<br/>유기물 등의 오염물질을 산화제로 산화 분해시켜 정화하는 데 소비되는 산소량을 의미하며, BOD와 함께 물의 오염된 정도를 표시하는 지표로 사용됨</p>
<p class="guide_txt">※ 총유기탄소량(TOC, Total Organic Carbon)란?<br/>유기물질 측정지표 중 하나로 시료를 고온으로 태우는 등의 방법으로 발생하는 이산화탄소를 측정하여 탄소량으로 표현</p>
<p class="guide_txt">※ 부유물질(SS, Suspended Solids)란?<br/>입자 지름이2mm 이하로 물에 용해되지 않는 물질을 일컫는 말로 오염된 물의 수질을 표시하는 지표</p>
<p class="guide_txt">※ 총질소(T-N, Total Nitrogen)란?<br/>무기성 질소 및 유기성 질소의 총량을 말하며 하천, 호소 등의 부영양화를 나타내는 지표의 하나</p>
<p class="guide_txt">※ 총인(T-P, Total Phosphorus)란?<br/>하천, 호소 등의 부영양화를 나타내는 지표의 하나로 물속에 포함된 인의 총량</p>
